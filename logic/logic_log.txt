> proc read_file { fname } {
  set lines {}
  if { ! [file exists $fname] } {
    return $lines
  }
  set fp [open $fname]
  set is_pio false
  set skip_empty false
  while { [gets $fp line] >= 0 } {
    if { [string first "# pio_begin" $line] == 0 } {
      set is_pio true
    } elseif { [string first "# pio_end" $line] == 0 } {
      set is_pio false
      set skip_empty true
    } elseif { ! $is_pio } {
      if { ! ($skip_empty && [regexp -expanded {^[ #]*$} $line]) } {
        lappend lines $line
        set skip_empty false
      }
    }
  }
  close $fp
  return $lines
}
